* 상품테이블 product
num / 순번 / number, PK, NN
pname / 상품이름, varchar2(50), NN
pprice / 상품가격, number, NN
pjaego / 재고, number, NN
pcompany 제조사, varchar2(50), NN

CREATE TABLE product (
    num number NOT NULL,
    pname varchar2(50) NOT NULL,
    pprice number NOT NULL,
    pjaego number NOT NULL,
    pcompany varchar(50) NOT NULL,
    PRIMARY KEY(num)
);
