CREATE TABLE member (
no INT NOT NULL AUTO_INCREMENT,
id VARCHAR(50) NOT NULL,
pwd VARCHAR(50) NOT NULL,
name VARCHAR(50) NOT NULL,
phone VARCHAR(50) NOT NULL,
job VARCHAR(50) NOT NULL,
PRIMARY KEY(id),
UNIQUE(no)
);
